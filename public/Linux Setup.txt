Magnet Link Cleaner (Alternative to https://addons.mozilla.org/en-US/firefox/addon/trackerless-magnets/ )
```
// ==UserScript==
// @name Magnet Cleaner
// ==/UserScript==

$('a[href^="magnet:"]').each(function() {
    var $button = $(this);
    var href = $button.attr('href');
    if (href)
        $button.attr('href', href.replace(/&.*$/, ''));
});
```
